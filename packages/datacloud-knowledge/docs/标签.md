```python
class FlexibleTagSystem:
    """
    灵活的标签系统
    支持属性标签、关系标签、类型标签的混合使用
    """
    
    def __init__(self, graph: MetadataGraph):
        self.graph = graph
        
    def add_tag(self, target_term_id: str, tag: dict, tag_type: str = "auto"):
        """
        为术语添加标签
        
        tag_type:
        - "property": 属性标签（方式1）
        - "relation": 关系标签（方式2）
        - "auto": 自动选择最佳方式
        """
        
        if tag_type == "property" or (tag_type == "auto" and self._should_be_property(tag)):
            return self._add_property_tag(target_term_id, tag)
        
        else:
            return self._add_relation_tag(target_term_id, tag)
    
    def _should_be_property(self, tag: dict) -> bool:
        """判断是否应该作为属性标签"""
        # 简单值（字符串、数字、日期等）
        isinstance(tag.get("value"), (str, int, float, bool))
        # TODO
    
    def _add_property_tag(self, term_id: str, tag: dict):
        """添加属性标签"""
        term_node = f"Term:{term_id}"
        
        # 直接修改节点的properties
        if term_node in self.graph.graph:
            current_props = self.graph.graph.nodes[term_node].get("properties", {})
            if isinstance(current_props, Properties):
                current_props = current_props.to_dict()
            
            current_props[tag["key"]] = tag["value"]
            
            # 更新节点
            self.graph.graph.nodes[term_node]["properties"] = Properties(current_props)
            
            return {
                "type": "property",
                "term_id": term_id,
                "key": tag["key"],
                "value": tag["value"]
            }
    
    def _add_relation_tag(self, term_id: str, tag: dict):
        """添加关系标签"""
        # 创建或获取目标Term
        target_term_id = tag.get("target_term")
        if not target_term_id:
            # 自动创建Term
            target_term_id = f"tag_{tag['key']}_{hash(tag['value'])}"
            target_term = TermNode(
                id=target_term_id,
                standard_name=tag["value"],
                properties=Properties(tag.get("target_properties", {}))
            )
            self.graph.add_term(target_term)
        
        # 创建关系
        relation_type = tag.get("relation_type", EdgeLabel.BELONGS_TO)
        properties = tag.get("properties", {})
        
        self.graph.add_term_relation(
            term_id,
            target_term_id,
            relation_type,
            properties=properties
        )
        
        return {
            "type": "relation",
            "source": term_id,
            "target": target_term_id,
            "relation": relation_type.value,
            "properties": properties
        }
```

查询时的灵活处理
```python
class FlexibleQuery:
    """支持灵活标签的查询"""
    
    def __init__(self, graph: MetadataGraph, tag_system: FlexibleTagSystem):
        self.graph = graph
        self.tag_system = tag_system
    
    def query_by_tag(self, tag_name: str, tag_value: Any = None) -> list:
        """
        按标签查询，自动处理标签类型
        
        示例：
        - query_by_tag("岗位", "销售") → 找到所有岗位为销售的员工
        - query_by_tag("评优结果") → 找到所有有评优结果的
        """
        results = []
        
        # 1. 查找属性标签
        for node_id, data in self.graph.graph.nodes(data=True):
            if data.get("node_type") == NodeLabel.TERM.value:
                props = data.get("properties", {})
                if isinstance(props, Properties):
                    props = props.to_dict()
                
                if tag_name in props:
                    if tag_value is None or props[tag_name] == tag_value:
                        results.append({
                            "term": data,
                            "tag_type": "property",
                            "match": props[tag_name]
                        })
        
        # 2. 查找关系标签
        for node_id, data in self.graph.graph.nodes(data=True):
            if data.get("node_type") == NodeLabel.TERM.value:
                term_id = data["id"]
                
                # 查找 outgoing edges
                for _, target, edge_data in self.graph.graph.out_edges(node_id, data=True):
                    target_data = self.graph.graph.nodes[target]
                    
                    # 检查是否是标签关系
                    if edge_data.get("relation") in [
                        EdgeLabel.BELONGS_TO.value,
                        EdgeLabel.HAS_PROPERTY.value
                    ]:
                        target_name = target_data.get("standard_name", "")
                        if tag_value is None or tag_value in target_name:
                            results.append({
                                "term": data,
                                "tag_type": "relation",
                                "relation": edge_data.get("relation"),
                                "target": target_data,
                                "match": target_name
                            })
        
        return results
    
    def get_all_tags(self, term_id: str) -> dict:
        """获取术语的所有标签（合并类型）"""
        term_node = f"Term:{term_id}"
        
        if term_node not in self.graph.graph:
            return {}
        
        data = self.graph.graph.nodes[term_node]
        
        tags = {
            "properties": {},      # 属性标签
            "relations": [],       # 关系标签
        }
        
        # 1. 属性标签
        props = data.get("properties", {})
        if isinstance(props, Properties):
            props = props.to_dict()
        tags["properties"] = props
        
        # 2. 关系标签
        for _, target, edge_data in self.graph.graph.out_edges(term_node, data=True):
            target_data = self.graph.graph.nodes[target]
            tags["relations"].append({
                "relation_type": edge_data.get("relation"),
                "target_id": target_data.get("id"),
                "target_name": target_data.get("standard_name"),
                "properties": edge_data
            })

        return tags
```

关键设计要点
1. 无模式灵活度
- 属性标签：随意添加，无约束
- 关系标签：自动创建目标Term，无需预定义
2. 自动类型推断
```
# 系统自动选择最佳标签类型
tag_system.add_tag("wangxm", {"key": "年龄", "value": 28})  
# → 自动识别为 property
```
3. 查询统一性
无论标签是如何存储的，查询接口统一：
```
# 查找"销售"标签，无论是属性还是关系
results = query.query_by_tag("销售")
```
4. 优先级策略
当同一个概念以多种形式存在时：
