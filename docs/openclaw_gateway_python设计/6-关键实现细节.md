# 6. 关键实现细节

## 6.1 与 deepagents 集成

```python
from deepagents import create_deep_agent
from langchain.chat_models import init_chat_model
from langgraph.checkpoint.base import BaseCheckpointSaver

def create_agent_from_config(
    config: AgentConfig,
    tenant_context: Optional[TenantContext] = None,
    provider_override: Optional[str] = None,
    model_override: Optional[str] = None,
    skills_entries: Optional[List[Dict]] = None,
    checkpointer: Optional[BaseCheckpointSaver] = None
) -> CompiledStateGraph:
    """
    根据配置创建 deepagents Agent (统一API)
    
    这是创建Agent的统一入口，整合:
    - 系统提示词构建
    - 子智能体配置
    - 租户上下文
    - 模型覆盖
    - Checkpoint支持 (用于STEER模式)
    """
    # 解析模型 (支持覆盖)
    model_str = model_override or config.model
    if "/" in model_str:
        provider, model_name = model_str.split("/", 1)
    else:
        provider, model_name = "anthropic", model_str
    
    # 构建系统提示词
    if config.custom_system_prompt:
        system_prompt = config.custom_system_prompt
    else:
        skills_prompt = _format_skills(skills_entries) if skills_entries else None
        
        builder = SystemPromptBuilder(
            workspace_dir=config.workspace_dir,
            config=SystemPromptConfig(
                prompt_mode=config.system_prompt_mode,
                bootstrap_max_chars=config.bootstrap_max_chars,
                enable_skills=config.enable_skills,
                enable_heartbeat=config.enable_heartbeat,
                user_timezone=config.user_timezone,
            )
        )
        
        system_prompt = builder.build(
            agent_id=config.id,
            model=f"{provider}:{model_name}",
            skills_prompt=skills_prompt,
            tool_names=config.tools,
            tenant_id=tenant_context.tenant_id if tenant_context else None
        )
    
    # 创建 deepagents Agent
    llm = init_chat_model(f"{provider}:{model_name}")
    
    # 转换子智能体配置
    subagents = _build_subagents_config(config.subagents)
    
    agent = create_deep_agent(
        model=llm,
        system_prompt=system_prompt,
        tools=config.tools,
        subagents=subagents,
        checkpointer=checkpointer  # 用于STEER模式
    )
    
    return agent


def _build_subagents_config(subagents: List[SubAgentConfig]) -> List[Dict]:
    """构建 deepagents 格式的子智能体配置"""
    result = []
    for sub in subagents:
        sub_config = {
            "name": sub.name,
            "description": sub.description,
            "system_prompt": sub.system_prompt,
        }
        if sub.model:
            sub_config["model"] = sub.model
        if sub.tools:
            sub_config["tools"] = sub.tools
        result.append(sub_config)
    return result


def _format_skills(skills: List[Dict]) -> str:
    """格式化skills为XML格式"""
    if not skills:
        return ""
    
    lines = ["<available_skills>"]
    for skill in skills:
        lines.append("  <skill>")
        lines.append(f"    <name>{skill.get('name', '')}</name>")
        lines.append(f"    <description>{skill.get('description', '')}</description>")
        lines.append(f"    <location>{skill.get('filePath', '')}</location>")
        lines.append("  </skill>")
    lines.append("</available_skills>")
    
    return "\n".join(lines)
```

class EventType(Enum):
    MESSAGE_START = "message_start"
    MESSAGE_DELTA = "message_delta"      # 流式文本
    MESSAGE_END = "message_end"
    TOOL_START = "tool_execution_start"
    TOOL_END = "tool_execution_end"
    AGENT_END = "agent_end"
    ERROR = "error"

@dataclass
class Event:
    type: EventType
    payload: Dict[str, Any]
    timestamp: datetime
