## 10.1 队列类型定义 (`queue/types.py`)

```python
from enum import Enum
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Callable, Any
import time


class QueueMode(Enum):
    """
    队列模式 - 对应 OpenClaw 的 QueueMode
    
    基于 LangGraph 的 interrupt/Command(resume) 机制实现
    """
    STEER = "steer"              # 立即注入当前运行 (使用 LangGraph interrupt)
    FOLLOWUP = "followup"        # 当前运行结束后处理
    COLLECT = "collect"          # 合并多条消息 (默认)
    STEER_BACKLOG = "steer-backlog"  # steer + followup: 立即注入 + 保留队列
    INTERRUPT = "interrupt"      # 中止当前运行
    QUEUE = "queue"              # 通用队列 (steer 的别名)


class DropPolicy(Enum):
    """溢出策略"""
    OLD = "old"                  # 丢弃最旧的
    NEW = "new"                  # 丢弃最新的
    SUMMARIZE = "summarize"      # 保留摘要 (默认)


@dataclass
class QueueSettings:
    """队列配置"""
    mode: QueueMode = QueueMode.COLLECT
    debounce_ms: int = 1000       # 防抖等待时间 (ms)
    cap: int = 20                 # 队列最大容量
    drop_policy: DropPolicy = DropPolicy.SUMMARIZE


@dataclass
class QueuedMessage:
    """队列中的消息项 - 简化版 (无 channel 概念)"""
    prompt: str
    session_key: str
    message_id: Optional[str] = None
    summary_line: Optional[str] = None
    enqueued_at: float = field(default_factory=time.time)
    
    # Agent 配置 (简化)
    agent_id: Optional[str] = None
    provider_override: Optional[str] = None
    model_override: Optional[str] = None


@dataclass
class QueueState:
    """队列状态 - 对应 OpenClaw 的 FollowupQueueState"""
    items: List[QueuedMessage] = field(default_factory=list)
    draining: bool = False
    last_enqueued_at: float = 0
    mode: QueueMode = QueueMode.COLLECT
    debounce_ms: int = 1000
    cap: int = 20
    drop_policy: DropPolicy = DropPolicy.SUMMARIZE
    dropped_count: int = 0
    summary_lines: List[str] = field(default_factory=list)
```

## 10.2 去重缓存 (`infra/dedupe.py`)

```python
import time
from typing import Dict, Optional


class DedupeCache:
    """
    TTL + LRU 去重缓存 - 对应 OpenClaw 的 createDedupeCache
    
    用于:
    - 入站消息去重 (防止重复处理)
    - 队列消息去重
    """
    
    def __init__(self, ttl_ms: int, max_size: int):
        self.ttl_ms = ttl_ms
        self.max_size = max_size
        self._cache: Dict[str, float] = {}
    
    def _cleanup(self):
        """清理过期条目"""
        now = time.time() * 1000
        expired = [
            k for k, v in self._cache.items() 
            if now - v > self.ttl_ms
        ]
        for k in expired:
            del self._cache[k]
    
    def check(self, key: str) -> bool:
        """
        检查键是否存在，存在则刷新 TTL
        返回: True 表示已存在 (重复)
        """
        self._cleanup()
        if key in self._cache:
            self._cache[key] = time.time() * 1000
            return True
        
        # 不存在，添加新条目
        if len(self._cache) >= self.max_size:
            # LRU: 移除最旧的
            oldest = min(self._cache.items(), key=lambda x: x[1])
            del self._cache[oldest[0]]
        
        self._cache[key] = time.time() * 1000
        return False
    
    def peek(self, key: str) -> bool:
        """仅检查是否存在，不刷新 TTL"""
        return key in self._cache
    
    def clear(self):
        """清空缓存"""
        self._cache.clear()


# 全局去重缓存实例
inbound_dedupe = DedupeCache(ttl_ms=20*60*1000, max_size=5000)  # 20分钟
queue_dedupe = DedupeCache(ttl_ms=5*60*1000, max_size=10000)    # 5分钟
```

## 10.3 队列状态管理 (`queue/manager.py`)

```python
import asyncio
from typing import Dict, Optional
from .types import QueueState, QueueSettings


class QueueManager:
    """
    队列状态管理器 - 对应 OpenClaw 的 FOLLOWUP_QUEUES Map
    
    管理所有会话的队列状态，按 session_key 隔离
    """
    
    DEFAULT_DEBOUNCE_MS = 1000
    DEFAULT_CAP = 20
    
    def __init__(self):
        self._queues: Dict[str, QueueState] = {}
        self._locks: Dict[str, asyncio.Lock] = {}
    
    def get_existing(self, key: str) -> Optional[QueueState]:
        """获取已存在的队列"""
        return self._queues.get(key.strip())
    
    def get_or_create(self, key: str, settings: QueueSettings) -> QueueState:
        """获取或创建队列"""
        key = key.strip()
        
        if key in self._queues:
            existing = self._queues[key]
            self._apply_settings(existing, settings)
            return existing
        
        # 创建新队列
        created = QueueState(
            mode=settings.mode,
            debounce_ms=settings.debounce_ms if settings.debounce_ms > 0 else self.DEFAULT_DEBOUNCE_MS,
            cap=settings.cap if settings.cap > 0 else self.DEFAULT_CAP,
            drop_policy=settings.drop_policy,
        )
        self._queues[key] = created
        self._locks[key] = asyncio.Lock()
        return created
    
    def _apply_settings(self, state: QueueState, settings: QueueSettings):
        """应用运行时设置更新"""
        if settings.mode:
            state.mode = settings.mode
        if settings.debounce_ms is not None:
            state.debounce_ms = max(0, settings.debounce_ms)
        if settings.cap is not None:
            state.cap = max(1, settings.cap)
        if settings.drop_policy:
            state.drop_policy = settings.drop_policy
    
    def clear_queue(self, key: str) -> int:
        """清空队列，返回清除的项目数"""
        key = key.strip()
        queue = self._queues.get(key)
        if not queue:
            return 0
        
        cleared = len(queue.items) + queue.dropped_count
        del self._queues[key]
        if key in self._locks:
            del self._locks[key]
        return cleared
    
    def get_lock(self, key: str) -> asyncio.Lock:
        """获取队列锁"""
        if key not in self._locks:
            self._locks[key] = asyncio.Lock()
        return self._locks[key]
```

## 10.4 消息入队 (`queue/enqueue.py`)

```python
from typing import Optional
from .types import QueuedMessage, QueueSettings, QueueState, DropPolicy
from .manager import QueueManager
from ..infra.dedupe import queue_dedupe


class MessageEnqueuer:
    """
    消息入队器 - 对应 OpenClaw 的 enqueueFollowupRun
    
    处理消息入队逻辑:
    - 去重检查
    - 溢出策略应用
    - 触发排空
    """
    
    def __init__(self, queue_manager: QueueManager):
        self.queue_manager = queue_manager
    
    def _build_dedupe_key(self, message: QueuedMessage, queue_key: str) -> Optional[str]:
        """构建去重键 (简化版)"""
        if not message.message_id:
            return None
        return f"queue:{queue_key}:{message.message_id}"
    
    def _is_duplicate(self, message: QueuedMessage, items: list) -> bool:
    """检查队列内是否重复 (简化版 - 仅检查 message_id 或 prompt)"""
        if message.message_id:
            return any(
            item.message_id == message.message_id
                for item in items
            )
        # 无 message_id 时，检查相同 session_key + prompt
        return any(
            item.session_key == message.session_key and
            item.prompt == message.prompt
            for item in items
        )
    
    def _apply_drop_policy(self, queue: QueueState, message: QueuedMessage) -> bool:
        """
        应用溢出策略
        返回: True 表示应该入队，False 表示丢弃
        """
        if len(queue.items) < queue.cap:
            return True
        
        if queue.drop_policy == DropPolicy.OLD:
            # 丢弃最旧的
            dropped = queue.items.pop(0)
            if dropped.summary_line:
                queue.summary_lines.append(dropped.summary_line)
            queue.dropped_count += 1
            return True
        
        elif queue.drop_policy == DropPolicy.NEW:
            # 丢弃最新的
            return False
        
        elif queue.drop_policy == DropPolicy.SUMMARIZE:
            # 保留摘要
            if message.summary_line:
                queue.summary_lines.append(message.summary_line)
            queue.dropped_count += 1
            return False
        
        return True
    
    async def enqueue(
        self,
        queue_key: str,
        message: QueuedMessage,
        settings: QueueSettings
    ) -> bool:
        """
        入队消息
        
        返回:
            True: 成功入队
            False: 被去重或丢弃
        """
        queue = self.queue_manager.get_or_create(queue_key, settings)
        lock = self.queue_manager.get_lock(queue_key)
        
        async with lock:
            # 1. 全局去重检查
            dedupe_key = self._build_dedupe_key(message, queue_key)
            if dedupe_key and queue_dedupe.peek(dedupe_key):
                return False
            
            # 2. 队列内去重
            if self._is_duplicate(message, queue.items):
                return False
            
            # 3. 应用溢出策略
            if not self._apply_drop_policy(queue, message):
                return False
            
            # 4. 入队
            queue.items.append(message)
            queue.last_enqueued_at = time.time()
            
            if dedupe_key:
                queue_dedupe.check(dedupe_key)
            
            return True
```

## 10.5 队列排空 (`queue/drain.py`)

```python
import asyncio
from typing import Callable, List
from .types import QueuedMessage, QueueState, QueueMode
from .manager import QueueManager


class QueueDrainer:
    """
    队列排空器 - 对应 OpenClaw 的 scheduleFollowupDrain
    
    后台任务，自动排空队列中的消息
    """
    
    def __init__(self, queue_manager: QueueManager):
        self.queue_manager = queue_manager
        self._drain_callbacks: dict = {}
        self._draining_tasks: dict = {}
    
    def register_callback(
        self,
        key: str,
        callback: Callable[[QueuedMessage], asyncio.Future]
    ):
        """注册排空回调"""
        self._drain_callbacks[key] = callback
    
    def kick_if_idle(self, key: str):
        """如果空闲则触发排空"""
        queue = self.queue_manager.get_existing(key)
        if queue and not queue.draining and key in self._drain_callbacks:
            task = asyncio.create_task(
                self._drain(key, self._drain_callbacks[key])
            )
            self._draining_tasks[key] = task
    
    async def _wait_debounce(self, queue: QueueState):
        """等待防抖周期"""
        await asyncio.sleep(queue.debounce_ms / 1000)
    
    def _build_collect_prompt(self, items: List[QueuedMessage], summary: str) -> str:
        """构建合并提示 - 对应 OpenClaw 的 buildCollectPrompt"""
        lines = ["[Queued messages while agent was busy]"]
        
        if summary:
            lines.append(f"[Earlier messages summarized: {summary}]")
        
        for i, item in enumerate(items, 1):
            lines.append(f"---\nQueued #{i}\n{item.prompt}")
        
        return "\n".join(lines)
    
    def _build_summary(self, queue: QueueState) -> str:
        """构建摘要"""
        if not queue.summary_lines:
            return ""
        return "; ".join(queue.summary_lines[:5])
    
    async def _drain(self, key: str, callback: Callable[[QueuedMessage], asyncio.Future]):
        """排空队列主循环"""
        queue = self.queue_manager.get_existing(key)
        if not queue or queue.draining:
            return
        
        queue.draining = True
        
        try:
            while queue.items or queue.dropped_count > 0:
                # 等待防抖
                await self._wait_debounce(queue)
                
                lock = self.queue_manager.get_lock(key)
                async with lock:
                    if not queue.items and not queue.dropped_count:
                        break
                    
                    if queue.mode == QueueMode.COLLECT:
                        await self._drain_collect_mode(key, queue, callback)
                    else:
                        await self._drain_individual_mode(key, queue, callback)
        
        except Exception as e:
            print(f"Queue drain failed for {key}: {e}")
            queue.last_enqueued_at = time.time()
        
        finally:
            queue.draining = False
            if not queue.items and not queue.dropped_count:
                self.queue_manager.clear_queue(key)
            if key in self._draining_tasks:
                del self._draining_tasks[key]
    
    async def _drain_collect_mode(self, key: str, queue: QueueState, callback):
        """COLLECT 模式: 合并处理"""
        if not queue.items:
            return
        
        items = list(queue.items)
        summary = self._build_summary(queue)
        
        combined_prompt = self._build_collect_prompt(items, summary)
        last_message = items[-1]
        
        combined_message = QueuedMessage(
            prompt=combined_prompt,
            session_key=last_message.session_key,
            originating_channel=last_message.originating_channel,
            originating_to=last_message.originating_to,
            agent_id=last_message.agent_id,
        )
        
        await callback(combined_message)
        
        # 清空队列
        queue.items.clear()
        queue.summary_lines.clear()
        queue.dropped_count = 0
    
    async def _drain_individual_mode(self, key: str, queue: QueueState, callback):
        """逐个处理模式"""
        if not queue.items:
            return
        
        message = queue.items.pop(0)
        await callback(message)
```

## 10.6 入站防抖 (`inbound/debounce.py`)

```python
import asyncio
from typing import Dict, List, Optional, Callable, TypeVar
from dataclasses import dataclass

T = TypeVar('T')


@dataclass
class DebounceBuffer:
    """防抖缓冲区"""
    items: List
    timeout: Optional[asyncio.Handle]
    debounce_ms: int


class InboundDebouncer:
    """
    入站消息防抖器 - 对应 OpenClaw 的 createInboundDebouncer
    
    合并同一发送者的快速连续消息
    """
    
    def __init__(
        self,
        debounce_ms: int,
        build_key: Callable[[T], Optional[str]],
        on_flush: Callable[[List[T]], asyncio.Future],
        should_debounce: Optional[Callable[[T], bool]] = None
    ):
        self.default_debounce_ms = debounce_ms
        self.build_key = build_key
        self.on_flush = on_flush
        self.should_debounce = should_debounce or (lambda x: True)
        self._buffers: Dict[str, DebounceBuffer] = {}
    
    async def _flush_buffer(self, key: str, buffer: DebounceBuffer):
        """刷新缓冲区"""
        if key in self._buffers:
            del self._buffers[key]
        
        if buffer.timeout:
            buffer.timeout.cancel()
        
        if buffer.items:
            await self.on_flush(buffer.items)
    
    def _schedule_flush(self, key: str, buffer: DebounceBuffer):
        """调度刷新"""
        if buffer.timeout:
            buffer.timeout.cancel()
        
        loop = asyncio.get_event_loop()
        buffer.timeout = loop.call_later(
            buffer.debounce_ms / 1000,
            lambda: asyncio.create_task(self._flush_buffer(key, buffer))
        )
    
    async def enqueue(self, item: T) -> bool:
        """
        入队消息
        
        返回:
            True: 消息被缓冲
            False: 消息立即处理
        """
        key = self.build_key(item)
        can_debounce = self.should_debounce(item)
        
        if not can_debounce or not key:
            # 立即处理
            if key and key in self._buffers:
                await self._flush_buffer(key, self._buffers[key])
            await self.on_flush([item])
            return False
        
        existing = self._buffers.get(key)
        if existing:
            existing.items.append(item)
            self._schedule_flush(key, existing)
            return True
        
        # 创建新缓冲区
        buffer = DebounceBuffer(
            items=[item],
            timeout=None,
            debounce_ms=self.default_debounce_ms
        )
        self._buffers[key] = buffer
        self._schedule_flush(key, buffer)
        return True
```

## 10.7 队列策略决策 (`queue/policy.py`)

```python
from enum import Enum
from .types import QueueMode


class QueueAction(Enum):
    """队列动作决策结果"""
    RUN_NOW = "run-now"           # 立即运行
    ENQUEUE_FOLLOWUP = "enqueue-followup"  # 入队等待
    DROP = "drop"                 # 丢弃


class QueuePolicy:
    """
    队列策略决策器 - 对应 OpenClaw 的 resolveActiveRunQueueAction
    
    根据当前状态和队列模式决定如何处理新消息
    """
    
    @staticmethod
    def resolve(
        is_active: bool,          # 是否有运行中的 Agent
        is_heartbeat: bool,       # 是否是心跳消息
        should_followup: bool,    # 是否应该 followup
        queue_mode: QueueMode     # 当前队列模式
    ) -> QueueAction:
        """
        解析队列动作
        
        决策逻辑:
        1. 无运行中 → 立即执行
        2. 心跳消息 → 丢弃
        3. steer 模式或 should_followup → 入队
        4. 否则 → 立即执行
        """
        if not is_active:
            return QueueAction.RUN_NOW
        
        if is_heartbeat:
            return QueueAction.DROP
        
        if queue_mode in (QueueMode.STEER, QueueMode.STEER_BACKLOG):
            return QueueAction.ENQUEUE_FOLLOWUP
        
        if should_followup:
            return QueueAction.ENQUEUE_FOLLOWUP
        
        return QueueAction.RUN_NOW
```

## 10.8 集成到 AgentRunner (`runner/queued_runner.py`)

```python
from typing import Dict, Optional
import asyncio
from ..queue import (
    QueueManager, MessageEnqueuer, QueueDrainer, 
    QueuedMessage, QueueSettings, QueueMode, QueuePolicy, QueueAction
)
from ..inbound.debounce import InboundDebouncer
from ..infra.dedupe import inbound_dedupe


class QueuedAgentRunner:
    """
    带队列支持的 Agent 运行器
    
    整合:
    - 入站消息去重
    - 防抖合并
    - 队列管理
    - Agent 执行
    """
    
    def __init__(self, agent_registry, session_manager, config):
        self.agent_registry = agent_registry
        self.session_manager = session_manager
        self.config = config
        
        # 初始化队列系统
        self.queue_manager = QueueManager()
        self.enqueuer = MessageEnqueuer(self.queue_manager)
        self.drainer = QueueDrainer(self.queue_manager)
        
        # 初始化防抖器
        self.debouncer = InboundDebouncer(
            debounce_ms=config.messages.inbound.debounce_ms,
            build_key=lambda msg: f"{msg.get('from')}:{msg.get('session_key')}",
            on_flush=self._handle_debounced_messages
        )
        
        # 跟踪运行中的任务
        self._running_tasks: Dict[str, asyncio.Task] = {}
    
    async def handle_message(
        self,
        session_key: str,
        prompt: str,
        message_id: Optional[str] = None,
        **kwargs
    ) -> dict:
        """
        处理入站消息 (主入口)
        
        完整流程:
        1. 去重检查
        2. 防抖处理
        3. 队列决策
        4. 执行或入队
        """
        # 1. 去重检查
        dedupe_key = f"{session_key}:{message_id}" if message_id else None
        if dedupe_key and inbound_dedupe.check(dedupe_key):
            return {"status": "deduped", "message": "Duplicate message"}
        
        # 2. 防抖处理
        message = {"prompt": prompt, "session_key": session_key, "from": kwargs.get("from")}
        debounced = await self.debouncer.enqueue(message)
        if debounced:
            return {"status": "debouncing", "message": "Message buffered"}
        
        # 3. 队列决策
        is_active = session_key in self._running_tasks
        queue_mode = self._resolve_queue_mode(session_key)
        
        action = QueuePolicy.resolve(
            is_active=is_active,
            is_heartbeat=False,
            should_followup=False,
            queue_mode=queue_mode
        )
        
        # 4. 执行或入队
        if action == QueueAction.RUN_NOW:
            return await self._execute_run(session_key, prompt, **kwargs)
        
        elif action == QueueAction.ENQUEUE_FOLLOWUP:
            return await self._enqueue_message(session_key, prompt, queue_mode, **kwargs)
        
        else:  # DROP
            return {"status": "dropped", "message": "Message dropped"}
    
    async def _enqueue_message(
        self,
        session_key: str,
        prompt: str,
        queue_mode: QueueMode,
        **kwargs
    ) -> dict:
        """入队消息 (简化版)"""
        message = QueuedMessage(
            prompt=prompt,
            session_key=session_key,
            message_id=kwargs.get("message_id"),
            agent_id=kwargs.get("agent_id")
        )
        
        settings = QueueSettings(mode=queue_mode)
        success = await self.enqueuer.enqueue(session_key, message, settings)
        
        if success:
            # 注册排空回调
            self.drainer.register_callback(
                session_key,
                lambda msg: self._execute_run(session_key, msg.prompt, **kwargs)
            )
            # 触发排空
            self.drainer.kick_if_idle(session_key)
            
            return {"status": "queued", "message": "Message queued"}
        else:
            return {"status": "dropped", "message": "Queue full or duplicate"}
    
    async def _execute_run(self, session_key: str, prompt: str, **kwargs) -> dict:
        """执行 Agent 运行"""
        session = self.session_manager.get(session_key)
        
        agent = self.agent_registry.create_deep_agent(
            agent_id=session.agent_id,
            provider_override=session.provider_override,
            model_override=session.model_override
        )
        
        task = asyncio.create_task(self._run_agent(agent, prompt))
        self._running_tasks[session_key] = task
        
        try:
            result = await task
            return result
        finally:
            if session_key in self._running_tasks:
                del self._running_tasks[session_key]
    
    async def _run_agent(self, agent, prompt: str) -> dict:
        """运行 Agent"""
        messages = [{"role": "user", "content": prompt}]
        full_response = []
        
        async for event in agent.astream({"messages": messages}):
            full_response.append(event)
        
        text = "".join(
            str(e.get("content", e.get("text", ""))) 
            for e in full_response
        )
        
        return {"status": "completed", "text": text}
    
    async def _handle_debounced_messages(self, messages: list):
        """处理防抖后的批量消息"""
        if not messages:
            return
        
        # 合并消息
        combined_prompt = "\n".join(m["prompt"] for m in messages)
        session_key = messages[0]["session_key"]
        
        await self.handle_message(session_key, combined_prompt)
    
    def _resolve_queue_mode(self, session_key: str) -> QueueMode:
        """解析队列模式 (可从配置或会话元数据获取)"""
        # 默认使用配置中的模式
        mode_str = self.config.messages.queue.mode
        return QueueMode(mode_str)
```

**完整队列实现要点总结**:

1. **队列系统**: 完整支持 6 种队列模式 (STEER/STEER_BACKLOG/COLLECT/FOLLOWUP/INTERRUPT/QUEUE)
2. **STEER 模式**: 基于 LangGraph interrupt/Command(resume) 实现消息注入
3. **去重机制**: TTL + LRU 缓存，基于 message_id 或 session_key+prompt 去重
4. **防抖合并**: 批量合并快速连续消息，按 session_key 分组
5. **并发安全**: 使用 asyncio.Lock 保护队列状态
6. **与 deepagents 集成**: 在 LangGraph 之上构建队列层，不影响 Agent 执行
